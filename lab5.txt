#include <iostream>
#include <sstream>
#include <string>
using namespace std;

bool check(char str[])
{
    int len=sizeof(str)/sizeof(str[0]);

    if (str[0] < 'A' || str[0] > 'Z'){
		return false;}

    if (str[len - 1] != '.'){
		return false;}
    int prev_state = 0;
    int curr_state = 0;
    int i;
    for(i=1;i<len;i++)
    {
        if (str[i] >= 'A' && str[i] <= 'Z'){
			curr_state = 0;}
        else if (str[i] == ' '){
			curr_state = 1;}
        else if (str[i] >= 'a' && str[i] <= 'z'){
			curr_state = 2;}
        else if (str[i] == '.'){
			curr_state = 3;}

        if (prev_state == curr_state && curr_state != 2){
			return false;}

        if (prev_state == 2 && curr_state == 0){
			return false;}
        if (curr_state == 3 && prev_state != 1){
			return (str[i + 1] == '\0');}

        prev_state = curr_state;
    }
return false;
}


int main()
{

    char input[100];
    int t=0;
    int i;
    int f1=0;

    cout<<"Enter input:\n";
    cin>>input;

    if(check(input)==true){
        cout<<"correct input.";
    }
    else{
    cout<<"incorrect input.";}


    return 0;
}
